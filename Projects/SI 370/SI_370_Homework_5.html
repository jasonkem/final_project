import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split, GridSearchCV
from sklearn.metrics import accuracy_score
from sklearn.ensemble import GradientBoostingClassifier, VotingClassifier
from sklearn.neural_network import MLPClassifier
from sklearn.tree import DecisionTreeClassifier
import os

train_df = pd.read_csv("/Users/jason/Desktop/SI 370 Master/si370fa23-student-main 2/data/train_hw5.csv")
test_df = pd.read_csv("/Users/jason/Desktop/SI 370 Master/si370fa23-student-main 2/data/test_hw5.csv")

print(train_df.describe())

print(train_df.info())

for col in train_df.columns:
    print(f"Value counts for {col}:\n{train_df[col].value_counts()}\n")

corr_matrix = train_df.select_dtypes(include=['int64', 'float64']).corr()
sns.heatmap(corr_matrix, annot=True, cmap='coolwarm')
plt.show()

train_df.hist(figsize=(12, 10))
plt.show()

passenger_ids = test_df['PassengerId'].copy()

for col in train_df.columns:
    if train_df[col].dtype == 'object' or train_df[col].dtype == 'bool':
        train_df[col].fillna(train_df[col].mode()[0], inplace=True)
    else:
        train_df[col].fillna(train_df[col].median(), inplace=True)

for col in test_df.columns:
    if test_df[col].dtype == 'object' or test_df[col].dtype == 'bool':
        test_df[col].fillna(test_df[col].mode()[0], inplace=True)
    else:
        test_df[col].fillna(test_df[col].median(), inplace=True)

y = train_df['Transported']
train_df.drop(['Transported'], axis=1, inplace=True)

train_df.drop(['PassengerId'], axis=1, inplace=True)
test_df.drop(['PassengerId'], axis=1, inplace=True)

train_df = pd.get_dummies(train_df, drop_first=True)
test_df = pd.get_dummies(test_df, drop_first=True)

train_df, test_df = train_df.align(test_df, join='inner', axis=1)

X_train, X_val, y_train, y_val = train_test_split(train_df, y, test_size=0.2, random_state=42)

param_grid = {
    'n_estimators': [100, 150, 200],
    'learning_rate': [0.05, 0.1, 0.2],
    'max_depth': [3, 4, 5]
}
gb = GradientBoostingClassifier(random_state=42)
gb_grid_search = GridSearchCV(estimator=gb, param_grid=param_grid, cv=3, n_jobs=-1, verbose=2, scoring='accuracy')
gb_grid_search.fit(X_train, y_train)
gb_best = gb_grid_search.best_estimator_

mlp = MLPClassifier(random_state=42)

dt = DecisionTreeClassifier(random_state=42)

voting_clf = VotingClassifier(
    estimators=[('gb', gb_best), ('mlp', mlp), ('dt', dt)],
    voting='hard'
)

voting_clf.fit(X_train, y_train)

y_val_pred = voting_clf.predict(X_val)

val_accuracy = accuracy_score(y_val, y_val_pred)
print("Validation Accuracy with Voting Classifier:", val_accuracy)

predictions = voting_clf.predict(test_df)

submission_df = pd.DataFrame({'PassengerId': passenger_ids, 'Transported': predictions.astype(bool)})

directory = '/Users/jason/Desktop/SI 370 Master/si370fa23-student-main 2/data'
filename = 'spaceship_titanic_submission_voting.csv'
submission_file_path = os.path.join(directory, filename)

submission_df.to_csv(submission_file_path, index=False)

print(f"Voting Classifier Submission file saved: {submission_file_path}")
